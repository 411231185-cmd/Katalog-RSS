def clean_brands(text: str) -> str:
if not isinstance(text, str):
return text
out = text
for brand in FORBIDDEN_BRANDS:
out = re.sub(re.escape(brand), "", out, flags=re.IGNORECASE) # чистим лишние пробелы
out = re.sub(r"\s{2,}", " ", out).strip()
return out

def make_short_description(name: str, article: str) -> str:
"""
Простое стандартное описание, без длинной кинематики.
"""
name = (name or "").strip()
article = (article or "").strip()

    parts = []
    if name:
        parts.append(f"{name} — запасная часть для токарного станка.")
    if article:
        parts.append(f"Артикул: {article}.")
    parts.append("Купить с доставкой по России.")
    parts.append("Подбор по каталогу и чертежам заказчика.")
    parts.append("ТД РУССтанкоСбыт.")

    return " ".join(parts)

def main():
print("Загружаю:", IN_OUT_PATH)
df = pd.read_excel(IN_OUT_PATH, engine="openpyxl")

    if NAME_COL not in df.columns or ARTICLE_COL not in df.columns:
        raise ValueError(f"Проверь NAME_COL ({NAME_COL}) и ARTICLE_COL ({ARTICLE_COL}) — таких колонок нет в файле.")
    if DESC_COL not in df.columns:
        df[DESC_COL] = ""

    changed = 0

    for idx, row in df.iterrows():
        name = row.get(NAME_COL, "")
        article = row.get(ARTICLE_COL, "")

        short_desc = make_short_description(name, article)
        short_desc = clean_brands(short_desc)

        new_desc = short_desc + "\n\n" + SERVICE_BLOCK

        df.at[idx, DESC_COL] = new_desc
        changed += 1

    print("Обновлено описаний:", changed)
    df.to_excel(IN_OUT_PATH, index=False, engine="openpyxl")
    print("Файл перезаписан:", IN_OUT_PATH)

if **name** == "**main**":
main()
