# EQUIPNET-clean.ps1
param([string]$File)

$lines = Get-Content $File -Encoding UTF8

$keep = foreach ($line in $lines) {
    $t = $line.Trim()

    if ([string]::IsNullOrWhiteSpace($t)) { continue }

    if ($t -match '\d+\s*руб') { continue }
    if ($t -match 'Добавлено:|Обновлено:|Просмотров:|В наличии|Под заказ') { continue }
    if ($t -match '^\d{2}\.\d{2}\.\d{4}$') { continue }

    if ($t.Length -gt 80) { continue }

    if ($t -match '^[A-Za-zА-Яа-яЁё0-9][A-Za-zА-Яа-яЁё0-9\s\.,\-/()]+$') {
        $t
    }
}

$unique = @()
$seen = @{}
foreach ($line in $keep) {
    $n = $line -replace '\s+', ' '
    if (-not $seen.ContainsKey($n)) {
        $seen[$n] = $true
        $unique += $n
    }
}

Set-Content -Path $File -Value $unique -Encoding UTF8
Write-Host "Готово: для $File"
