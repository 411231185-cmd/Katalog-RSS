param(
[Parameter(Mandatory = $true)]
[string]$File
)

$lines = Get-Content $File -Encoding UTF8

$skipWords = @(
'Опубликовано',
'Редактировать',
'Сохранить',
'Удалить',
'Подробнее',
'Источник',
'Показать',
'Скрыть',
'Читать далее',
'Перейти'
)

$offerLines = foreach ($line in $lines) {
$t = $line.Trim()

    if ([string]::IsNullOrWhiteSpace($t)) { continue }
    if ($t -match '^\d{2}\.\d{2}\.\d{4}$') { continue }
    if ($t -match '^#') { continue }
    if ($t -match '^```') { continue }
    if ($skipWords -contains $t) { continue }

    $t = $t -replace '^-+\s*', ''
    $t = $t -replace '^\s*[\*\+]\s*', ''
    $t = $t.Trim()

    if ([string]::IsNullOrWhiteSpace($t)) { continue }

    if ($t -match '^[A-Za-zА-Яа-яЁё0-9][A-Za-zА-Яа-яЁё0-9\s\.,\-()№"«»/]+$') {
        $t
    }

}

$offerLines = $offerLines | Select-Object -Unique

Set-Content -Path $File -Value $offerLines -Encoding UTF8

Write-Host "Готово: оставлены только названия офферов в файле $File"
