#Requires -Version 5.1
<#
.SYNOPSIS
Clean markdown/text files with offer lists - remove descriptions, duplicates, etc.

.DESCRIPTION
Removes descriptions, duplicates, prices, dates, phones, statuses, and other noise.
Keeps only product/offer names.

.PARAMETER FilePath
Path to the file to clean (required)

.PARAMETER Mode
Cleaning mode: 'Strict' (default) or 'Soft'
Strict = maximum cleaning, Soft = minimal cleaning

.PARAMETER Backup
Create backup file before cleaning (default: $true)

.EXAMPLE
.\clean-offers.ps1 -FilePath "C:\data\offers.txt"
.\clean-offers.ps1 -FilePath "C:\data\offers.txt" -Mode Soft
.\clean-offers.ps1 -FilePath "C:\data\offers.txt" -Mode Strict -Backup $false
#>

[CmdletBinding()]
param(
[Parameter(Mandatory = $true)]
[string]$FilePath,

    [Parameter(Mandatory = $false)]
    [ValidateSet('Strict', 'Soft')]
    [string]$Mode = 'Strict',

    [Parameter(Mandatory = $false)]
    [bool]$Backup = $true

)

Set-StrictMode -Version Latest
$ErrorActionPreference = 'Stop'

# ============================================================================

# CONFIGURATION

# ============================================================================

Write-Host ""
Write-Host ("╔" + ("=" _ 76) + "╗")
Write-Host ("║ OFFER LIST CLEANER - Mode: $Mode".PadRight(77) + "║")
Write-Host ("╚" + ("=" _ 76) + "╝")
Write-Host ""

# Check file exists

if (-not (Test-Path $FilePath)) {
Write-Host "ERROR: File not found: $FilePath" -ForegroundColor Red
exit 1
}

Write-Host "[*] Reading file: $FilePath" -ForegroundColor Cyan

# Read file

try {
[string[]]$allLines = @(Get-Content -Path $FilePath -Encoding UTF8 -ErrorAction Stop)
}
catch {
Write-Host "ERROR: Cannot read file: $\_" -ForegroundColor Red
exit 1
}

$totalLines = $allLines.Count
Write-Host "[+] Loaded: $totalLines lines" -ForegroundColor Green

# ============================================================================

# FILTER PATTERNS

# ============================================================================

# Words that indicate description text

$descriptionWords = @(
'предназначен', 'предназначена', 'используется', 'используются',
'применяется', 'применяются', 'служит', 'обеспечивает', 'обеспечивают',
'конструкция', 'характеристики', 'описание', 'запасная часть', 'запасные части',
'подбор и изготовление', 'изготовление', 'строим', 'выполнена',
'в условиях интенсивного', 'для визуального контроля', 'для регулирования',
'для передачи', 'для перемещения', 'для крепления', 'для защиты', 'для фильтрации',
'поставляется как отдельно', 'в сборе', 'этапы производства', 'в т.ч',
'может использоваться', 'предназначены', 'особенности', 'преимущества',
'спецификация', 'требование', 'рекомендация', 'инструкция', 'руководство'
)

# Phrases that indicate service text

$serviceWords = @(
'опубликовано', 'редактировать', 'сохранить', 'удалить', 'подробнее',
'источник', 'показать', 'скрыть', 'читать далее', 'перейти', 'смотреть',
'купить', 'заказать', 'добавить в', 'корзину', 'кнопка', 'ссылка',
'мобильное приложение', 'скачать', 'оцените товар', 'рейтинг'
)

# Company names and plan names

$companyWords = @(
'лтд', 'llc', 'ооо', 'зао', 'пао', 'ао', 'компания', 'фирма', 'предприятие',
'завод', 'фабрика', 'производство', 'оптовая', 'розничная', 'дистрибьютор',
'поставщик', 'тариф', 'тарифный план', 'мобила', 'москва', 'санкт-петербург'
)

# ============================================================================

# HELPER FUNCTIONS

# ============================================================================

function Is-DescriptionLine {
param([string]$Text)

    $lower = $Text.ToLower()

    # Check for description words
    foreach ($word in $descriptionWords) {
        if ($lower -match "\b$word\b") {
            return $true
        }
    }

    return $false

}

function Is-ServiceLine {
param([string]$Text)

    $lower = $Text.ToLower()

    # Check for service words
    foreach ($word in $serviceWords) {
        if ($lower -match "\b$word\b") {
            return $true
        }
    }

    return $false

}

function Is-CompanyLine {
param([string]$Text)

    $lower = $Text.ToLower()

    foreach ($word in $companyWords) {
        if ($lower -match "\b$word\b") {
            return $true
        }
    }

    return $false

}

function Is-PriceLine {
param([string]$Text)

    # Prices with currency symbols
    if ($Text -match '[₽$€£¥₹₽][0-9\s,.-]+') {
        return $true
    }

    # Prices with text currency
    if ($Text -match '\d+\s*(руб|рублей|долл|евро|usd|eur|грн)') {
        return $true
    }

    # Long numbers that look like prices
    if ($Text -match '^\s*\d+\s*([,.])\s*\d{2}') {
        return $true
    }

    return $false

}

function Is-PhoneLine {
param([string]$Text)

    # Phone patterns
    if ($Text -match '\+?7\s*[\(\-]?\d{3}[\)\-]?\d{3}[\-]?\d{2}[\-]?\d{2}') {
        return $true
    }

    if ($Text -match '\+?[\d\s\(\)\-\+]{10,}') {
        return $true
    }

    if ($Text -match 'тел:|phone:|tel:|звонить|позвонить') {
        return $true
    }

    return $false

}

function Is-DateLine {
param([string]$Text)

    # DD.MM.YYYY
    if ($Text -match '^\d{2}\.\d{2}\.\d{4}$') {
        return $true
    }

    # DD.MM.YY
    if ($Text -match '^\d{2}\.\d{2}\.\d{2}$') {
        return $true
    }

    # Dates in text
    if ($Text -match '\d{1,2}\s(января|февраля|марта|апреля|мая|июня|июля|августа|сентября|октября|ноября|декабря)\s\d{4}') {
        return $true
    }

    return $false

}

function Is-StockLine {
param([string]$Text)

    $lower = $Text.ToLower()

    if ($lower -match 'наличие:|в наличии|под заказ|нет в наличии|выпущен|отправлен|доступен') {
        return $true
    }

    return $false

}

function Is-MarkdownHeaderOrCode {
param([string]$Text)

    if ($Text -match '^#+\s') {
        return $true
    }

    if ($Text -match '^```') {
        return $true
    }

    return $false

}

function Is-LongDescription {
param([string]$Text, [int]$MaxLength = 150)

    # If very long and contains description patterns, it's probably a description
    if ($Text.Length -gt $MaxLength) {
        if ($Text -match '[,;:]\s+[а-яё]') {
            return $true
        }
    }

    return $false

}

function Is-ValidOfferName {
param([string]$Text)

    # Must have at least some letters or numbers
    if ($Text -match '[A-Za-zА-Яа-яЁё0-9]') {
        # Should NOT be only whitespace or special chars
        if ($Text -match '[A-Za-zА-Яа-яЁё0-9]{2,}') {
            return $true
        }
    }

    return $false

}

# ============================================================================

# STAGE 1: INITIAL CLEANING

# ============================================================================

Write-Host "[*] Stage 1: Initial cleaning..." -ForegroundColor Cyan

$cleaned = @()

foreach ($line in $allLines) {
$text = $line.Trim()

    # Skip empty lines
    if ([string]::IsNullOrWhiteSpace($text)) {
        continue
    }

    # Remove markdown list markers
    $text = $text -replace '^\s*[-*+]\s+', ''
    $text = $text -replace '^\s*\d+\.\s+', ''
    $text = $text.Trim()

    # Skip empty after cleanup
    if ([string]::IsNullOrWhiteSpace($text)) {
        continue
    }

    # Skip markdown headers and code blocks
    if (Is-MarkdownHeaderOrCode -Text $text) {
        continue
    }

    # Skip service lines
    if (Is-ServiceLine -Text $text) {
        continue
    }

    # Skip stock/availability lines
    if (Is-StockLine -Text $text) {
        continue
    }

    # Skip price lines
    if (Is-PriceLine -Text $text) {
        continue
    }

    # Skip phone lines
    if (Is-PhoneLine -Text $text) {
        continue
    }

    # Skip date lines
    if (Is-DateLine -Text $text) {
        continue
    }

    # Skip company lines
    if (Is-CompanyLine -Text $text) {
        continue
    }

    # Skip description lines
    if (Is-DescriptionLine -Text $text) {
        continue
    }

    # In Strict mode, skip very long descriptions
    if ($Mode -eq 'Strict') {
        if (Is-LongDescription -Text $text -MaxLength 120) {
            continue
        }
    }

    # Check if it looks like a valid offer name
    if (Is-ValidOfferName -Text $text) {
        $cleaned += $text
    }

}

$afterStage1 = $cleaned.Count
Write-Host "[+] After stage 1: $afterStage1 lines" -ForegroundColor Green

# ============================================================================

# STAGE 2: REMOVE DUPLICATES (preserve order)

# ============================================================================

Write-Host "[*] Stage 2: Removing duplicates..." -ForegroundColor Cyan

$unique = @()
$seen = @{}

foreach ($line in $cleaned) {
$lower = $line.ToLower()

    if (-not $seen.ContainsKey($lower)) {
        $unique += $line
        $seen[$lower] = $true
    }

}

$afterStage2 = $unique.Count
$duplicateCount = $afterStage1 - $afterStage2

Write-Host "[+] After stage 2: $afterStage2 lines (removed $duplicateCount duplicates)" -ForegroundColor Green

# ============================================================================

# STAGE 3: ADDITIONAL STRICT FILTERING (if Strict mode)

# ============================================================================

if ($Mode -eq 'Strict') {
Write-Host "[*] Stage 3: Strict mode additional filtering..." -ForegroundColor Cyan

    $strict = @()

    foreach ($line in $unique) {
        # Skip if contains too many numbers (probably an ID or something)
        if ($line -match '^\d+$') {
            continue
        }

        # Skip if contains only special characters
        if ($line -match '^[^A-Za-zА-Яа-яЁё0-9]+$') {
            continue
        }

        # Skip very short lines (1-2 chars)
        if ($line.Length -lt 3) {
            continue
        }

        $strict += $line
    }

    $afterStage3 = $strict.Count
    $strictRemoved = $afterStage2 - $afterStage3

    Write-Host "[+] After stage 3: $afterStage3 lines (removed $strictRemoved in strict filter)" -ForegroundColor Green

    $final = $strict

}
else {
$final = $unique
}

# ============================================================================

# SAVE RESULTS

# ============================================================================

Write-Host ""
Write-Host "[*] Saving results..." -ForegroundColor Cyan

# Create backup

if ($Backup) {
    $backupPath = "$FilePath.backup.txt"
Copy-Item -Path $FilePath -Destination $backupPath -Force
Write-Host "[+] Backup created: $backupPath" -ForegroundColor Green
}

# Save cleaned file

$final | Out-File -FilePath $FilePath -Encoding UTF8 -Force
Write-Host "[+] File saved: $FilePath" -ForegroundColor Green

# ============================================================================

# STATISTICS

# ============================================================================

Write-Host ""
Write-Host ("╔" + ("=" _ 76) + "╗")
Write-Host ("║ CLEANING STATISTICS".PadRight(77) + "║")
Write-Host ("╠" + ("=" _ 76) + "╣")
Write-Host ("║ Original lines: " + $totalLines.ToString().PadRight(49) + "║").Substring(0, 80)
Write-Host ("║ After stage 1: " + $afterStage1.ToString().PadRight(49) + "║").Substring(0, 80)
Write-Host ("║ Duplicates removed: " + $duplicateCount.ToString().PadRight(49) + "║").Substring(0, 80)

if ($Mode -eq 'Strict') {
Write-Host ("║ Strict filter removed: " + $strictRemoved.ToString().PadRight(49) + "║").Substring(0, 80)
}

Write-Host ("║ FINAL LINES: " + $final.Count.ToString().PadRight(49) + "║").Substring(0, 80)

$compressionRatio = [Math]::Round((1 - ($final.Count / $totalLines)) * 100, 1)
Write-Host ("║  Compression ratio:     " + "$compressionRatio%".PadRight(49) + "║").Substring(0, 80)

Write-Host ("╚" + ("=" \* 76) + "╝")

Write-Host ""
Write-Host "[+] DONE! File cleaned successfully." -ForegroundColor Green
Write-Host ""
