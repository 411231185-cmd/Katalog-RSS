python
import re
from pathlib import Path

import pandas as pd

# Корень репозитория

ROOT = Path(r"C:\GitHub-Repositories\Katalog-RSS").resolve()
STANKI_DIR = ROOT / "docs" / "Станки"
TKP_DIR = ROOT / "docs" / "ТКП"

# Путь к нормализованному каталогу; если у тебя другое имя файла —

# просто поменяй на свой.

CATALOG_CSV = ROOT / "catalogs" / "datanormalizedcatalogwithtildaurls.csv"

CONTACTS_BLOCK = """

## Контактная информация

Телефон: +7 499 390-85-04  
Email: zakaz@tdrusstankosbyt.ru  
Telegram: @tdrusstankosbyt  
Сайт: https://tdrusstankosbyt.ru
""".strip()

def load_catalog():
if not CATALOG_CSV.exists():
print(f"Каталог не найден: {CATALOG_CSV}")
return {}

    df = pd.read_csv(CATALOG_CSV, encoding="utf-8-sig", engine="python")
    # Нормализуем SKU → model (16k20, 1m63n и т.п.)
    sku_col = None
    for cand in ["SKU", "sku", "Sku", "Article"]:
        if cand in df.columns:
            sku_col = cand
            break
    if not sku_col:
        print("Не найден столбец SKU в каталоге")
        return {}

    df[sku_col] = df[sku_col].astype(str).str.strip().str.lower()

    # Попробуем вытащить полезные столбцы, но не падаем, если чего‑то нет
    title_col = next((c for c in df.columns if c.lower() == "title"), None)
    desc_col = next((c for c in df.columns if c.lower() in ("description", "desc", "fulldescription")), None)
    price_col = next((c for c in df.columns if "price" in c.lower()), None)
    url_col = next((c for c in df.columns if "url" in c.lower()), None)

    catalog = {}
    for _, row in df.iterrows():
        sku = str(row[sku_col]).strip().lower()
        # Ожидаем, что SKU содержит модель как подстроку: 16k20, 1m63n и т.п.
        catalog[sku] = {
            "title": str(row[title_col]).strip() if title_col else "",
            "description": str(row[desc_col]).strip() if desc_col else "",
            "price": str(row[price_col]).strip() if price_col else "",
            "url": str(row[url_col]).strip() if url_col else "",
        }

    return catalog

def best_catalog_match(model: str, catalog: dict) -> dict:
"""
Ищем в каталоге запись, где SKU содержит код модели (16k20, 1m63n и т.п.).
"""
model_l = model.lower()
for sku, data in catalog.items():
if model_l in sku:
return data
return {}

def read_text(path: Path) -> str:
return path.read_text(encoding="utf-8")

def write_text(path: Path, text: str):
path.write_text(text, encoding="utf-8")

def split_md_sections(md: str) -> dict:
parts = re.split(r"\n(?=##\s+)", md)
out = {"**intro**": parts[0].strip()}
for chunk in parts[1:]:
m = re.match(r"##\s+(._)\n(._)\Z", chunk.strip(), flags=re.S)
if not m:
continue
title = m.group(1).strip()
body = m.group(2).strip()
out[title] = body
return out

def pick_stanki_block(stanki_md: str, keywords) -> str:
sections = split_md_sections(stanki_md)
for title, body in sections.items():
t = re.sub(r"\s+", " ", title.lower())
if any(k in t for k in keywords) and body:
return body.strip() # fallback: первый абзац интро
intro = sections.get("**intro**", "").strip()
if intro:
return intro
return ""

def to_bullets(text: str, max_items: int = 8) -> str: # режем по строкам/предложениям
raw = [x.strip(" -•\t") for x in re.split(r"\n+|(?<=[.!?])\s+", text) if x.strip()]
items = []
for x in raw:
if len(x) < 20:
continue
items.append(x)
if len(items) >= max_items:
break
if not items:
return ""
return "\n".join(f"- {i}" for i in items)

def replace_section_if_todo(md: str, header: str, new_block: str) -> str:
if not new_block.strip():
return md

    pattern = rf"(##\s+{re.escape(header)}\s*\n)(.*?)(\n##\s+|\Z)"
    m = re.search(pattern, md, flags=re.S)
    if not m:
        return md

    old_body = m.group(2)
    if "TODO" not in old_body:
        # уже что‑то есть, не трогаем
        return md

    return md[:m.start(2)] + new_block.strip() + md[m.end(2):]

def enrich_one_model(model_dir: Path, catalog: dict):
model = model_dir.name # например, "16k20"
readme = model_dir / "README.md"
stanki_file = STANKI_DIR / f"{model}.md"

    if not readme.exists():
        return False

    md = read_text(readme)

    stanki_md = read_text(stanki_file) if stanki_file.exists() else ""

    # 1) Описание станка
    if "## Описание станка" in md and "TODO" in md:
        block = pick_stanki_block(stanki_md, ["описание", "общее описание", "краткое"])
        if block:
            # берём 2–4 предложения
            desc = " ".join(re.split(r"(?<=[.!?])\s+", block)[:4]).strip()
            md = replace_section_if_todo(md, "Описание станка", desc)

    # 2) Назначение и возможности
    if "## Назначение и возможности" in md and "TODO" in md:
        block = pick_stanki_block(stanki_md, ["назначение", "возможности", "операции"])
        bullets = to_bullets(block)
        if bullets:
            md = replace_section_if_todo(md, "Назначение и возможности", bullets)

    # 3) Ключевые преимущества
    if "## Ключевые преимущества" in md and "TODO" in md:
        block = pick_stanki_block(stanki_md, ["преимущества", "особенности", "плюсы"])
        bullets = to_bullets(block)
        if bullets:
            md = replace_section_if_todo(md, "Ключевые преимущества", bullets)

    # 4) Блок "Купить / Цена / Оффер" из каталога
    cat_data = best_catalog_match(model, catalog)
    if cat_data and "url" in cat_data and cat_data["url"]:
        buy_pattern = r"(##\s+(Купить|Заказать|Цена)[^\n]*\n)(.*?)(\n##\s+|\Z)"
        m = re.search(buy_pattern, md, flags=re.S)
        if m and "TODO" in m.group(3):
            lines = []
            if cat_data.get("price"):
                lines.append(f"Ориентировочная цена: **{cat_data['price']}** руб.")
            lines.append(f"Оформить заказ и актуальную цену уточнить можно на посадочной странице: {cat_data['url']}")
            if cat_data.get("description"):
                lines.append("")
                lines.append(cat_data["description"])
            new_block = "\n".join(lines).strip()
            md = md[:m.start(3)] + new_block + md[m.end(3):]

    # 5) Контакты
    if "Контактная информация" in md and "TODO" in md:
        md = md.replace("TODO", CONTACTS_BLOCK)

    write_text(readme, md)
    return True

def main():
catalog = load_catalog()
if not catalog:
print("Каталог пустой или не загружен — заполним только из docs/Станки.")
changed = 0
total = 0
for model_dir in TKP_DIR.iterdir():
if not model_dir.is_dir():
continue
total += 1
if enrich_one_model(model_dir, catalog):
changed += 1
print(f"Обновлено ТКП: {changed}/{total}")

if **name** == "**main**":
main()

python
import re
from pathlib import Path
import pandas as pd

ROOT = Path(r"C:\GitHub-Repositories\Katalog-RSS").resolve()
STANKI_DIR = ROOT / "docs" / "Станки"
TKP_DIR = ROOT / "docs" / "ТКП"
CATALOG_CSV = ROOT / "catalogs" / "datanormalizedcatalogwithtildaurls.csv" # подставь свой путь

CONTACTS_BLOCK = """

## Контактная информация

Телефон: +7 499 390-85-04  
Email: zakaz@tdrusstankosbyt.ru  
Telegram: @tdrusstankosbyt  
Сайт: https://tdrusstankosbyt.ru
""".strip()

FALLBACK_DESC = "Токарный станок с ЧПУ/универсальный для обработки валов, втулок, фланцев. Полная номенклатура запчастей и узлов в наличии."
FALLBACK_LIST = "- Высокая точность и долговечность.\n- Совместимость с современными ЧПУ-системами.\n- Быстрая поставка ЗИП из Москвы."

def load*catalog():
if not CATALOG_CSV.exists():
print(f"CSV не найден: {CATALOG_CSV}")
return {}
df = pd.read_csv(CATALOG_CSV, encoding="utf-8-sig")
sku_col = next((c for c in df.columns if 'sku' in c.lower()), None)
if not sku_col:
return {}
df[sku_col] = df[sku_col].astype(str).str.lower().str.strip()
title_col = next((c for c in df.columns if 'title' in c.lower()), None)
desc_col = next((c for c in df.columns if any(k in c.lower() for k in ('desc', 'text'))), None)
price_col = next((c for c in df.columns if 'price' in c.lower()), None)
url_col = next((c for c in df.columns if 'url' in c.lower()), None)
catalog = {}
for *, row in df.iterrows():
sku = row[sku_col]
catalog[sku] = {
'title': str(row.get(title_col, '')),
'desc': str(row.get(desc_col, '')),
'price': str(row.get(price_col, '')),
'url': str(row.get(url_col, '')),
}
return catalog

def best_catalog_match(model, catalog):
model_l = model.lower()
for sku, data in catalog.items():
if model_l in sku:
return data
return {}

def pick_stanki_block(stanki_md, keywords):
sections = re.split(r'\n(?=##\s+)', stanki_md)
for chunk in sections:
m = re.match(r'##\s+(._)\n(._)\Z', chunk.strip(), re.S)
if m:
title, body = m.group(1).strip(), m.group(2).strip()
t_lower = re.sub(r'\s+', ' ', title.lower())
if any(k in t_lower for k in keywords) and len(body) > 50:
return body # Fallback: intro or any text with keywords
intro = sections[0].strip() if sections else ""
if any(k in intro.lower() for k in keywords):
return intro[:500] # Ultimate fallback: first 300 chars
return stanki_md[:300] if stanki_md else ""

def to_bullets(text, max_items=8):
sentences = re.split(r'(?<=[.!?])\s+', text)
items = [s.strip() for s in sentences if len(s.strip()) > 20][:max_items]
if items:
return '\n'.join(f"- {item}" for item in items)
return FALLBACK_LIST

def replace_or_fill_section(md, header, new_block):
pattern = rf"(##\s+{re.escape(header)}\s*\n)(.*?)(?=\n##\s+|\Z)"
m = re.search(pattern, md, re.S)
if not m:
return md
old_body = m.group(2).strip()
if len(old_body) > 50 or 'TODO' not in old_body:
return md # уже заполнено
filled = new_block.strip()
if not filled:
filled = FALLBACK_DESC if 'описание' in header.lower() else FALLBACK_LIST
return md[:m.start(2)] + filled + md[m.end(2):]

def enrich_model(model_dir, catalog):
model = model_dir.name
readme_path = model_dir / "README.md"
stanki_path = STANKI_DIR / f"{model}.md"
md = readme_path.read_text(encoding='utf-8')
stanki_md = stanki_path.read_text(encoding='utf-8') if stanki_path.exists() else ""
cat_data = best_catalog_match(model, catalog)

    # Описание
    desc_keywords = ['описание', 'общее', 'краткое', 'характеристика']
    desc_block = pick_stanki_block(stanki_md, desc_keywords) or cat_data.get('title', '') + ' ' + cat_data.get('desc', '')
    md = replace_or_fill_section(md, 'Описание станка', desc_block[:400])

    # Назначение
    nazn_keywords = ['назначение', 'возможности', 'операции', 'применение']
    nazn_block = to_bullets(pick_stanki_block(stanki_md, nazn_keywords) or cat_data.get('desc', ''))
    md = replace_or_fill_section(md, 'Назначение и возможности', nazn_block)

    # Преимущества
    prev_keywords = ['преимущества', 'особенности', 'плюсы', 'преимуществ']
    prev_block = to_bullets(pick_stanki_block(stanki_md, prev_keywords))
    md = replace_or_fill_section(md, 'Ключевые преимущества', prev_block)

    # Купить из каталога
    if cat_data.get('url'):
        buy_block = f"Заказать станок и запчасти: {cat_data['url']}\nЦена: {cat_data.get('price', 'по запросу')}"
        md = replace_or_fill_section(md, 'Купить', buy_block) or replace_or_fill_section(md, 'Цена', buy_block)

    # Контакты
    if 'Контактная информация' in md:
        md = re.sub(r'TODO.*?(?=\n##|$)', CONTACTS_BLOCK, md, flags=re.S | re.I)

    readme_path.write_text(md, encoding='utf-8')
    print(f"✓ {model}: описание={len(desc_block)>50}, преимущества={len(prev_block)>50}")
    return True

def main():
catalog = load_catalog()
changed = 0
for model_dir in TKP_DIR.iterdir():
if model_dir.is_dir():
if enrich_model(model_dir, catalog):
changed += 1
print(f"v2 завершено: {changed} моделей обновлено")

if **name** == "**main**":
main()
